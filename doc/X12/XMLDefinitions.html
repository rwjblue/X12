def initialize(str)
  doc = Document.new(str)
  definitions = doc.root.name =~ /^Definition$/ ? doc.root.elements.to_a : [doc.root]
  definitions.each { |element|
    #puts element.name

    syntax_element = case element.name
                     when /table/     then parse_table(element)
                     when /segment/   then parse_segment(element)
                     when /composite/ then parse_composite(element)
                     when /loop/      then parse_loop(element)
                     end

    self[syntax_element.class] ||= {}
    self[syntax_element.class][syntax_element.name] = syntax_element
  }
end
